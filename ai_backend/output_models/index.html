class ContextSummaryModel(BaseModel):
    """
    Represents the LLM-enriched understanding of the device context.
    This does not introduce new domain fields ‚Äî it only summarizes
    and interprets existing ones for downstream GSPR reasoning.
    """

    medical_device: Optional[str] = Field(..., description="Type or name of the medical device.")
    intended_purpose: Optional[str] = Field(..., description="Intended purpose as defined by the manufacturer.")
    device_type: Optional[list[str]] = Field(..., description="Type(s) of medical device (e.g., implantable, software-based).")
    material_type: Optional[list[str]] = Field(..., description="Material types involved in the device design.")

    reasoning_focus: Optional[str] = Field(..., description="LLM's interpretation of which aspects (components, risk rules, materials) should be prioritized for GSPR reasoning.")
